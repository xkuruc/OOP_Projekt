22       private int spocitajHodnotenie(ArrayList<Integer> hodnotenie){

24           for(Integer cislo : hodnotenie){
25               vysledok += cislo;
26           }
27           return vysledok;

29       private ImageEntity vyhodnotVyhercu(){
30           int max = 0;
31           ImageEntity vyherca = imageService.returnObjectArray().get(0);
32           for(ImageEntity image : imageService.returnObjectArray()){
33               int cislo = spocitajHodnotenie(image.getHodnotenie());
34               if(cislo>max){
35                   max = cislo;
36                   vyherca = image;
37               }
38           }
39           this.vyherca = vyherca;
40           this.vyhercaPocetBodov = max;
41           return vyherca;
42       }
