28   public class LoginUI extends BaseUI{
29       private VBox vbox;
30       private Scene scene;
31       private ToggleGroup toggleGroup;
32       private TextField nameField;
33       public LoginUI(){
34           this.vbox = new VBox();
35           this.scene = new Scene(vbox, Color.GRAY);
36       }
37   
38       public VBox getVbox(){
39           return this.vbox;
40       }
41       public Scene getScene(){
42           return this.scene;
43       }
44       @Override
45       protected void setupContent(Stage stage) {
46           stageSetup(stage, 400, 400);
47           stage.setScene(getScene());
48           createRadioButton();
49           createNameInput();
50   
51           createLoginButton(stage);
52   
53           stage.show();
54       }
55   
56       public void handle() {
57           try {
58               logIn(stage);
59           } catch (Chybajuce_meno | Chybajuca_rola | Chybajuce_udaje e ) {
60               Label label = new Label(e.getMessage());
61               vbox.getChildren().add(label);
62           }
63       }
64   
65       public void createRadioButton(){
66           ToggleGroup toggleGroup = new ToggleGroup();
67           RadioButton radioButton1 = new RadioButton("Profesional");
68           radioButton1.setToggleGroup(toggleGroup);
69   
70           RadioButton radioButton2 = new RadioButton("Amater");
71           radioButton2.setToggleGroup(toggleGroup);
72   
73           RadioButton radioButton3 = new RadioButton("Novy");
74           radioButton3.setToggleGroup(toggleGroup);
75           getVbox().getChildren().addAll(radioButton1, radioButton2, radioButton3);
76           this.toggleGroup= toggleGroup;
77       }
78       public void createLoginButton(Stage stage){
79           Button button = new Button("LOGIN");
80   
81           button.setOnAction(e -> {
82               handle();
83               //logIn(stage);
84           });
85           getVbox().getChildren().add(button);
86       }
87       public void createNameInput(){
88           TextField nameField = new TextField();
89           nameField.setPromptText("Prihlasovacie meno");
90           vbox.getChildren().add(nameField);
91           this.nameField = nameField;
92       }
93   
94       private void logIn(Stage stage) throws Chybajuce_meno, Chybajuca_rola, Chybajuce_udaje{
95           if(nameField.getText().isEmpty() && toggleGroup.getSelectedToggle() == null){
96               throw new Chybajuce_udaje("Vypln vsetky udaje!");
97           } else if(nameField.getText().isEmpty() ){
98               throw new Chybajuce_meno("Vloz meno !");
99           } else if (toggleGroup.getSelectedToggle() == null) {
100              throw new Chybajuca_rola("Vyber si rolu !");
101          } else{
102              stage.close();
103              Uzivatel uzivatel;
104              if( (uzivatel = userService.exists(nameField.getText().toString())) == null){
105                  uzivatel = Factory.createUserFactory(((RadioButton) toggleGroup.getSelectedToggle()).getText())
106                          .createUzivatel(nameField.getText().toString());
107                  userService.PridajObjekt(uzivatel);
108              }
109              //userService.updateData(uzivatel);
110              userService.setAktualnyPouzivatel(uzivatel);
111  
112              MainUI mainUI = new MainUI();
113              mainUI.MainUISetup();
114          }
115      }
116      @Override
117      protected String getTitle() {
118          return "LOGIN";
