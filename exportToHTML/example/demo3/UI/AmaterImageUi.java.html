42       public void addImage(ImageEntity imageEntity){
43   
44           Label label = createLabel(imageEntity.getAutor());
45           content.getChildren().add(label);
46           String url = "file:"+ Paths.get(getClass().getProtectionDomain().getCodeSource().getLocation().getPath()
47                   + "/example/demo3/obrasky", imageEntity.getUrl()).toString();
48   
49           ImageView imageView = createImageView(url);
50           content.getChildren().add(imageView);
51   
52           HBox hBox = new HBox();
53           VBox.setMargin(hBox,  new Insets(10, 0, 50, 150));
54           toggleGroups.add(createHodnotenie(hBox, imageEntity));
55   
56           content.getChildren().add(hBox);
57           getScrollPane().setContent(content);
58       }
59       public ToggleGroup createHodnotenie(HBox hBox, ImageEntity imageEntity){
60           ToggleGroup toggleGroup = new ToggleGroup();
61           for(int i=0; i<5; i++){
62               final int value = i + 1;
63               createButton(hBox, toggleGroup, String.valueOf(value), new EventHandler() {
64                   @Override
65                   public void handle() {
66                       updateHondnotenie(imageEntity , value);
67                       imageEntity.updateHodnotenie(userService.getAktualnyPouzivatel(), value);
68                   }
69               } );
70           }
71           int i;
72           if((i=userService.getAktualnyPouzivatel().getIndexOhodnotenia(imageEntity)) >=0){
73               int index = userService.getAktualnyPouzivatel().getHodnotenia().get(i);
74               toggleGroup.selectToggle((RadioButton) hBox.getChildren().get(index-1));
75           }
76           return toggleGroup;
77       }
78       private RadioButton createButton(HBox hBox, ToggleGroup toggleGroup, String text, EventHandler eventHandler){
79           RadioButton button =  new RadioButton(text);
80           button.setToggleGroup(toggleGroup);
81           hBox.setSpacing(30);
82           button.setStyle("-fx-font-size: 14px;");
83           button.setOnAction(event -> {
84               eventHandler.handle();
85           });
86           hBox.getChildren().add(button);
87           return button;
88       }
