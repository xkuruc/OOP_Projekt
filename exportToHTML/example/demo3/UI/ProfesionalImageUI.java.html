69       public ToggleGroup createHodnotenie(HBox hBox, ImageEntity imageEntity){
70           ToggleGroup toggleGroup = new ToggleGroup();
71   
72           for(int i=0; i<5; i++){
73               final int value = i + 1;
74               createButton(hBox, toggleGroup, String.valueOf(value), new EventHandler() {
75                   @Override
76                   public void handle() {
77                       ((Profesional) userService.getAktualnyPouzivatel()).vymazSuperHodnotenie();
78                       updateHondnotenie(imageEntity , value);
79                       if(superTogglegroup!=null){
80                           superTogglegroup.selectToggle(null);
81                           superImage.updateHodnotenie(userService.getAktualnyPouzivatel(), 0);
82                       }
83                       imageEntity.updateHodnotenie(userService.getAktualnyPouzivatel(), value);
84                   }
85               } );
86           }
87           int i;
88           if((i=userService.getAktualnyPouzivatel().getIndexOhodnotenia(imageEntity)) >=0){
89               int index = userService.getAktualnyPouzivatel().getHodnotenia().get(i);
90               toggleGroup.selectToggle((RadioButton) hBox.getChildren().get(index-1));
91           }
92           return toggleGroup;
93       }

95       public ToggleGroup createSuperHodnotenie(HBox hBox, ImageEntity imageEntity){
96           ToggleGroup toggleGroup = new ToggleGroup();
97           RadioButton upvoteButton = createButton(hBox, toggleGroup, "Upvote", new EventHandler() {
98               @Override
99               public void handle() {
100                  superHodnotenie(toggleGroup, imageEntity, hodnotaSuperHodnotenia);
101                  //imageEntity.pridajHodnotenie(hodnotaSuperHodnotenia);
102              }
103          });
104  
105          RadioButton downvoteButton = createButton(hBox, toggleGroup, "Downvote", new EventHandler() {
106              @Override
107              public void handle() {
108                  superHodnotenie(toggleGroup, imageEntity, -hodnotaSuperHodnotenia);
109                  //imageEntity.pridajHodnotenie(-hodnotaSuperHodnotenia);
110              }
111          });
112  
113          //if(((Profesional) userService.getAktualnyPouzivatel()).getSuperHodnotenie())
114          Profesional profesional =((Profesional) userService.getAktualnyPouzivatel());
115          if(profesional.getSuperHodnotenie()!=0 && profesional.getSuperHodnotenieImage().getUrl().equals(imageEntity.getUrl())){
116              if(profesional.getSuperHodnotenie()>0){
117                  toggleGroup.selectToggle(upvoteButton);
118              }else{
119                  toggleGroup.selectToggle(downvoteButton);
120              }
121              this.superTogglegroup = toggleGroup;
122          }
123  
124          return toggleGroup;
125      }

127      private RadioButton createButton(HBox hBox, ToggleGroup toggleGroup, String text, EventHandler eventHandler){
128          RadioButton button =  new RadioButton(text);
129          button.setToggleGroup(toggleGroup);
130          hBox.setSpacing(30);
131          button.setStyle("-fx-font-size: 14px;");
132          button.setOnAction(event -> {
133              eventHandler.handle();
134          });
135  
136          hBox.getChildren().add(button);
137          return button;
138      }
139      private void superHodnotenie(ToggleGroup toggleGroup, ImageEntity imageEntity, int hodnota){
140          Uzivatel user = userService.getAktualnyPouzivatel();
141          ((Profesional) user).setSuperHodnotenie(hodnota);
142          ((Profesional) user).setSuperHodnotenieImage(imageEntity);
143  
144          user.setHodnotenia(new ArrayList<>());
145          user.setOhodnoteneObrasky(new ArrayList<>());
146          for(ToggleGroup toggleGroup1 : toggleGroups){
147              toggleGroup1.selectToggle(null);
148          }
149          if(this.superTogglegroup != null && this.superTogglegroup != toggleGroup){
150              superTogglegroup.selectToggle(null);
151          }
152          this.superTogglegroup = toggleGroup;
153          for(ImageEntity image : imageService.returnObjectArray()){
154              image.updateHodnotenie(userService.getAktualnyPouzivatel(), 0);
155              if(image.getUrl().equals(imageEntity.getUrl())){
156                  imageEntity.updateHodnotenie(userService.getAktualnyPouzivatel(), hodnota);
157              }
158          }
159          superImage = imageEntity;
160      }
