4    import javafx.stage.Stage;

6    //aby si len vedel, keby si neiplmementoval tu UISetup interface, tak to je template design pattern
7    //ze zrobis si takuto abstraktnu nadtriedu, a jej deti iba prepisuju jej metody, a potom vytvoris instanciu podtriedy ale na BaseUI (na nadtriede),
8    // a zavolas funkciu, ktoru nemaju zadefinovanu, ako teraz setupUI, a sa zavola ich vlastna implementacia setupContent
9    public abstract class BaseUI extends DataServiceManager implements UISetup{
10       protected abstract void setupContent(Stage stage);
11       @Override
12       public void setupUI(Stage stage) {
13           setupContent(stage);
14       }
15   
16       @Override
17       public void stageSetup(Stage stage, int width, int height){
18           stage.setTitle(getTitle());
19           stage.setWidth(width);
20           stage.setHeight(height);
21           stage.setResizable(false);
22       }
23   
24       protected abstract String getTitle();
25   
26   }
