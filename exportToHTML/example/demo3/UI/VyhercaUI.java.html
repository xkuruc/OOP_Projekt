15   public class VyhercaUI extends BaseUI{
16       private VBox vBox1;
17       private VBox content;
18       private Scene scene;
19       private ImageEntity vyhernyObrazok;
20       private int vyhercaPocetBodov;
21   
22       public VyhercaUI(ImageEntity vyhernyObrazok, int vyhercaPocetBodov){
23           this.vBox1 = new VBox();
24           this.content = new VBox();
25           this.scene = new Scene(vBox1);
26           this.vyhernyObrazok = vyhernyObrazok;
27           this.vyhercaPocetBodov = vyhercaPocetBodov;
28       }
29       @Override
30       protected void setupContent(Stage stage) {
31           Stage stage1 = createNewWindow(500, 500);
32           addImage(vyhernyObrazok);
33           Label label = new Label("Pocet bodov: "+ vyhercaPocetBodov);
34           label.setStyle("-fx-font-size: 20px; -fx-font-weight: bold; -fx-padding: 10px 10px 0px 100px");
35           vBox1.getChildren().add(content);
36           vBox1.getChildren().add(label);
37           stage1.setScene(scene);
38           stage1.show();
39       }
40   
41       private Stage createNewWindow(int width, int height){
42           Stage stage = new Stage();
43           stageSetup(stage, width, height);
44           return stage;
45       }
46       private ImageView createImageView(String imageUrl, int width, int height){
47           Image image = new Image(imageUrl);
48           ImageView imageView = new ImageView(image);
49           imageView.setFitWidth(width);
50           imageView.setFitHeight(height);
51           return imageView;
52       }
53       private Label createLabel(String text){
54           Label label = new Label(text);
55           label.setStyle("-fx-font-size: 40px; -fx-font-weight: bold; -fx-padding: 10px 10px 0px 200px");
56           return label;
57       }
58       private void addImage(ImageEntity imageEntity){
59           VBox vboxAkozeVsetko = new VBox();
60           HBox hBox = new HBox();
61   
62           vboxAkozeVsetko.getChildren().add(createLabel(imageEntity.getAutor()));
63   
64           String url = "file:"+ Paths.get(imageService.getDestination(), imageEntity.getUrl());
65           ImageView imageView = createImageView(url, 300, 300);
66           hBox.getChildren().add(imageView);
67           VBox.setMargin(hBox,  new Insets(10, 0, 10, 100));
68   
69           vboxAkozeVsetko.getChildren().add(hBox);
70           content.getChildren().add(vboxAkozeVsetko);
71       }
72
